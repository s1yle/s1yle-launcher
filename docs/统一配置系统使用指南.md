## 1. 快速开始

### 1.1 配置系统初始化

配置系统在应用启动时自动初始化，无需手动调用：

```typescript
// src/main.tsx - 已自动处理
import { config } from './config';

async function initApp() {
  await invokeAccInit();           // 1. 账户初始化
  await config.initialize();       // 2. 配置初始化（核心）
  await Promise.all([              // 3. 其他 store 初始化
    useThemeStore.getState().init(),
    useInstanceStore.getState().init(),
    useDownloadStore.getState().init(),
  ]);
}
```

### 1.2 基础使用示例

```typescript
import { config } from '@/config';

// 获取配置值
const theme = config.getConfigValue('preferences.theme');
const downloadPath = config.getConfigValue('download.download_path');

// 设置配置值
await config.setConfigValue('preferences.theme', 'dark');
await config.setConfigValue('download.concurrent_limit', 32);
```

## 2. 配置访问方式

### 2.1 方式 1：使用 config 单例（推荐）

**适用于**：所有场景，尤其是非 React 环境

```typescript
import { config } from '@/config';

// 等待配置就绪
await config.whenReady();

// 获取配置值（类型安全）
const theme = config.getConfigValue('preferences.theme');
const language = config.getConfigValue('preferences.language');
const enableAnimation = config.getConfigValue('preferences.enable_animation');

// 设置配置值
await config.setConfigValue('preferences.theme', 'light');
await config.setConfigValue('preferences.language', 'en-US');

// 获取实例配置
const instanceConfig = config.getInstanceConfig('instance-uuid');
await config.updateInstanceConfig('instance-uuid', {
  memory: { max_memory: 4096 }
});

// 订阅配置变更
const unsubscribe = config.on('change', (key, value) => {
  console.log(`配置变更：${key} = ${value}`);
});

// 取消订阅
unsubscribe();
```

### 2.2 方式 2：使用 Hooks（React 组件中推荐）

**适用于**：React 组件内部

```typescript
import { useConfig, usePreferences, useInstanceConfig, useDownloadConfig } from '@/hooks/useConfig';

function SettingsPage() {
  // 全局配置
  const { config, loading, error } = useConfig();
  
  // 用户偏好
  const { preferences, setTheme, setLanguage, toggleAnimation } = usePreferences();
  
  // 实例配置
  const { instanceConfig, updateJava, updateMemory } = useInstanceConfig(instanceId);
  
  // 下载配置
  const { downloadConfig, setDownloadPath, setConcurrentLimit } = useDownloadConfig();
  
  if (loading) return <div>加载中...</div>;
  if (error) return <div>错误：{error}</div>;
  
  return (
    <div>
      <h1>设置</h1>
      
      {/* 主题设置 */}
      <select value={preferences?.theme} onChange={(e) => setTheme(e.target.value)}>
        <option value="dark">暗夜</option>
        <option value="light">晨曦</option>
      </select>
      
      {/* 内存设置 */}
      <input
        type="number"
        value={instanceConfig?.memory.max_memory}
        onChange={(e) => updateMemory(1024, Number(e.target.value))}
      />
    </div>
  );
}
```

## 3. 配置类型定义

### 3.1 配置键类型（ConfigKey）

```typescript
type ConfigKey = 
  // 用户偏好
  | 'preferences.theme'
  | 'preferences.accent_color'
  | 'preferences.language'
  | 'preferences.enable_animation'
  
  // 下载配置
  | 'download.download_path'
  | 'download.concurrent_limit'
  | 'download.auto_verify'
  
  // 窗口位置
  | 'window_position.x'
  | 'window_position.y'
  | 'window_position.width'
  | 'window_position.height'
  | 'window_position.maximized'
  
  // 实例配置
  | 'instance_configs'
  | 'known_folders';
```

### 3.2 配置值类型

```typescript
// 主题模式
type ThemeMode = 'dark' | 'light' | 'system';

// 强调色
type AccentColor = 
  | 'indigo'  // 靛蓝
  | 'blue'    // 蓝色
  | 'green'   // 绿色
  | 'purple'  // 紫色
  | 'red'     // 红色
  | 'orange'  // 橙色
  | 'pink';   // 粉色

// 语言
type Language = 'zh-CN' | 'en-US';

// 配置值类型推断
type ConfigValue<T extends ConfigKey> = 
  T extends 'preferences.theme' ? ThemeMode :
  T extends 'preferences.accent_color' ? AccentColor :
  T extends 'preferences.language' ? Language :
  T extends 'preferences.enable_animation' ? boolean :
  T extends 'download.download_path' ? string :
  T extends 'download.concurrent_limit' ? number :
  T extends 'download.auto_verify' ? boolean :
  T extends `window_position.${string}` ? number :
  T extends 'instance_configs' ? Record<string, InstanceConfig> :
  T extends 'known_folders' ? any[] :
  never;
```

### 3.3 使用示例

```typescript
// ✅ 类型安全：编译时发现错误
const theme: ThemeMode = config.getConfigValue('preferences.theme');
await config.setConfigValue('preferences.theme', 'dark');

// ❌ 编译错误：类型不匹配
await config.setConfigValue('preferences.theme', 'invalid'); // Error!
await config.setConfigValue('preferences.theme', 123); // Error!
```

## 4. 配置事件订阅

### 4.1 订阅配置就绪

```typescript
import { config } from '@/config';

// 等待配置就绪
await config.whenReady();
console.log('配置已就绪');

// 或者订阅就绪事件
config.on('ready', () => {
  console.log('配置已就绪，可以访问');
});
```

### 4.2 订阅配置变更

```typescript
// 订阅所有配置变更
const unsubscribeAll = config.on('change', (key, value) => {
  console.log(`配置变更：${key} = ${value}`);
});

// 订阅特定配置变更
const unsubscribeTheme = config.on('change', (key, value) => {
  if (key === 'preferences.theme') {
    console.log('主题变更为:', value);
    // 更新 UI
  }
});

// 订阅多个配置
const keys = ['preferences.theme', 'preferences.language'];
const unsubscribeMulti = config.on('change', (key, value) => {
  if (keys.includes(key)) {
    handleConfigChange(key, value);
  }
});

// 取消订阅
unsubscribeAll();
unsubscribeTheme();
```

### 4.3 在 React 组件中使用

```typescript
import { useEffect } from 'react';
import { config } from '@/config';

function ThemeComponent() {
  useEffect(() => {
    // 订阅配置变更
    const unsubscribe = config.on('change', (key, value) => {
      if (key === 'preferences.theme') {
        // 更新主题
        applyTheme(value);
      }
    });
    
    // 清理订阅
    return unsubscribe;
  }, []);
  
  return <div>...</div>;
}
```

### 4.4 订阅错误

```typescript
config.on('error', (error) => {
  console.error('配置系统错误:', error);
  // 显示错误提示
  showErrorToast('配置加载失败');
});
```

## 5. 实例配置管理

### 5.1 获取实例配置

```typescript
import { config } from '@/config';

const instanceId = 'uuid-123';
const instanceConfig = config.getInstanceConfig(instanceId);

if (instanceConfig) {
  console.log('实例名称:', instanceConfig.name);
  console.log('Java 路径:', instanceConfig.java.java_path);
  console.log('最大内存:', instanceConfig.memory.max_memory);
}
```

### 5.2 更新实例配置

```typescript
// 更新 Java 配置
await config.updateInstanceConfig(instanceId, {
  java: {
    java_path: 'C:/Program Files/Java/jdk-17/bin/java.exe',
    java_args: ['-XX:+UseG1GC', '-Xmx4G'],
    use_bundled: false
  }
});

// 更新内存配置
await config.updateInstanceConfig(instanceId, {
  memory: {
    min_memory: 1024,
    max_memory: 4096
  }
});

// 更新图形配置
await config.updateInstanceConfig(instanceId, {
  graphics: {
    width: 1920,
    height: 1080,
    fullscreen: true
  }
});

// 更新多个配置
await config.updateInstanceConfig(instanceId, {
  name: '生存服务器',
  icon_path: '/path/to/icon.png',
  custom_args: ['--tweakClass=com.example.Tweak']
});
```

### 5.3 删除实例配置

```typescript
await config.removeInstanceConfig(instanceId);
```

### 5.4 使用 Hook 管理实例配置

```typescript
import { useInstanceConfig } from '@/hooks/useConfig';

function InstanceSettings({ instanceId }) {
  const { instanceConfig, updateJava, updateMemory } = useInstanceConfig(instanceId);
  
  const handleSaveJava = async () => {
    try {
      await updateJava(javaPath, javaArgs);
      success('Java 配置保存成功');
    } catch (e) {
      error('保存失败', e.message);
    }
  };
  
  const handleSaveMemory = async () => {
    try {
      await updateMemory(minMemory, maxMemory);
      success('内存配置保存成功');
    } catch (e) {
      error('保存失败', e.message);
    }
  };
  
  return (
    <div>
      <h2>Java 设置</h2>
      <input value={javaPath} onChange={...} />
      <button onClick={handleSaveJava}>保存</button>
      
      <h2>内存设置</h2>
      <input type="number" value={maxMemory} onChange={...} />
      <button onClick={handleSaveMemory}>保存</button>
    </div>
  );
}
```

## 6. 最佳实践

### ✅ **推荐做法**

#### 1. 使用统一入口访问配置

```typescript
// ✅ 推荐
const theme = config.getConfigValue('preferences.theme');

// ❌ 避免
const config = useConfigStore.getState().config;
const theme = config?.preferences.theme;
```

#### 2. 等待配置就绪后再访问

```typescript
// ✅ 推荐
await config.whenReady();
const value = config.getConfigValue('key');

// ❌ 避免
const value = config.getConfigValue('key'); // 可能返回 undefined
```

#### 3. 使用类型安全的配置键

```typescript
// ✅ 推荐
config.getConfigValue('preferences.theme');

// ❌ 避免：字符串拼写错误无法在编译时发现
config.getConfigValue('prefernces.them');
```

#### 4. 订阅配置变更时清理

```typescript
// ✅ 推荐
useEffect(() => {
  const unsubscribe = config.on('change', handleConfigChange);
  return unsubscribe;
}, []);

// ❌ 避免：未清理导致内存泄漏
config.on('change', handleConfigChange);
```

#### 5. 错误处理

```typescript
// ✅ 推荐
try {
  await config.setConfigValue('preferences.theme', 'dark');
  success('设置保存成功');
} catch (e) {
  const msg = e instanceof Error ? e.message : '保存失败';
  error('保存失败', msg);
  logger.error('保存配置失败', e);
}

// ❌ 避免：不处理错误
await config.setConfigValue('preferences.theme', 'dark');
```

### ❌ **禁止做法**

#### 1. 禁止直接修改配置文件

```typescript
// ❌ 禁止：绕过配置系统
fs.writeFileSync(configPath, JSON.stringify(config));

// ✅ 推荐：使用配置 API
await config.setConfigValue('key', value);
```

#### 2. 禁止在配置未就绪时访问

```typescript
// ❌ 禁止
const theme = config.getConfigValue('preferences.theme');
if (theme === 'dark') { ... } // 可能为 undefined

// ✅ 推荐
await config.whenReady();
const theme = config.getConfigValue('preferences.theme') || 'dark';
```

#### 3. 禁止混用多种配置访问方式

```typescript
// ❌ 禁止：同一配置通过不同方式访问
const theme1 = config.getConfigValue('preferences.theme');
const theme2 = localStorage.getItem('preferences.theme');

// ✅ 推荐：统一使用 config API
const theme = config.getConfigValue('preferences.theme');
```

## 7. 常见问题

### Q1: 配置何时加载完成？

**A**: 配置在应用启动时自动加载，可以通过以下方式等待：

```typescript
await config.whenReady();
// 或者
config.on('ready', () => { ... });
```

### Q2: 配置修改何时生效？

**A**: 配置修改立即生效（内存中），并异步保存到磁盘（防抖 500ms）。

### Q3: 如何监听配置变化？

**A**: 使用 `config.on('change', callback)` 订阅：

```typescript
config.on('change', (key, value) => {
  if (key === 'preferences.theme') {
    applyTheme(value);
  }
});
```

### Q4: 配置文件在哪里？

**A**: 配置文件位于 `{base_dir}/.smcl/app_config.json`

### Q5: 如何重置配置？

**A**: 使用 `resetConfig` API：

```typescript
await useConfigStore.getState().resetConfig();
```

### Q6: 如何导入导出配置？

**A**: 使用配置导入导出功能：

```typescript
// 导出配置
await useConfigStore.getState().exportConfig('/path/to/backup.json');

// 导入配置
await useConfigStore.getState().importConfig('/path/to/backup.json');
```

### Q7: 实例配置和全局配置有什么区别？

**A**: 
- **全局配置**：影响整个应用（如主题、语言）
- **实例配置**：只影响特定游戏实例（如 Java 路径、内存）

### Q8: 如何获取所有实例配置？

**A**: 访问 `instance_configs` 键：

```typescript
const allConfigs = config.getConfigValue('instance_configs');
```
